```js
import fetch from 'node-fetch';

export async function handler () {
  const artists = await fetch('https://www.analogstudios.net/api/artists').then(resp => resp.json());

  return {
    statusCode: 200,
    body: `
      <!DOCTYPE html>
      <html lang="en">
        <body>
          <h1>Artists</h1>
          <table>
            <thead>
              <tr>
                <td>Name</td>
                <td>Image</td>
              </tr>
            </thead>
            <tbody>
              ${artists.map(artist => {
                return `
                  <tr>
                    <td>
                      <h2>${artist.name}</h2>
                    </td>
                    <td>
                      <img src="${artist.imageUrl}" alt="${artist.name}"/>
                    </td>
                  </tr>
                `;
              }).join('')}
            </tbody>
          </table>
        </body>
      </html>
      `
  };
}
```
